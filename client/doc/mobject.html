结构

// 坐标点
message MOBJECT_POSITION
{
    optional int32 x = 1; // 横坐标
    optional int32 y = 2; // 纵坐标
}
    
// 地图上的对象(除开道具)
message MOBJECT_OBJECT
{
    optional MOBJECT_POSITION position = 1; // 坐标
    optional int32 curHp = 2; // 当前血量
    optional int32 maxHp = 3; // 血量上限
    optional MOBJECT_MOVE moveInfo = 4; // 路径信息
    optional int32 level = 5; // 等级
}
    
// 移动信息
message MOBJECT_MOVE
{
    optional int32 speed = 1; // 每秒移动的像素
    repeated MOBJECT_POSITION path = 2; // 关键点路径
}
